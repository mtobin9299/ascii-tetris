Module Game_screen

module Game_screen: sig .. end

The abstract screen type

type t 
type pos = int * int

pos is the type that represents an x, y coord pair

type pixel_type =
| CharPixel of char
| UnicodePixel of string

pixel_type is the type representing the types of pixels that can be printed to the screen

type pixel = pixel_type * ANSITerminal.style list 
val init_screen : int -> int -> t

init_screen w h is a screen of width w and height h

val render_screen : t -> float -> Input.t -> Game_state.t -> unit

render_screen screen delta_time input st draws the current state st of the game as well as the rest of the screen to the terminal
